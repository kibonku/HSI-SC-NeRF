--data    \
--output-dir p   \
--sfm-tool colmap   --matching-method sequential   --feature-type any   --matcher-type any   \

    --pipeline.model.camera-optimizer.mode   \
    --pipeline.model.hsi_loss_mult  \
    --pipeline.model.angular_loss_mult  \
